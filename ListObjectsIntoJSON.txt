using System;
using System.Collections.Generic;
using Newtonsoft.Json;

class Person
{
    public string Name { get; set; }
    public int Age { get; set; }
    public string Email { get; set; }
}

class Program
{
    static void Main()
    {
        // Create a list of Person objects
        List<Person> people = new List<Person>
        {
            new Person { Name = "Bob", Age = 22, Email = "bob@gmail.com" },
            new Person { Name = "John", Age = 25, Email = "john@example.com" },
            new Person { Name = "Alice", Age = 28, Email = "alice@example.com" }
        };

        // Convert the list to a JSON array
        string jsonArray = JsonConvert.SerializeObject(people, Formatting.Indented);

        // Print JSON output
        Console.WriteLine(jsonArray);
    }
}
