using System;
using System.IO;
using Newtonsoft.Json;
using Newtonsoft.Json.Linq;
using System.Xml;

{
    static void Main()
    {

        string filePath = "C:\\Users\\khant\\OneDrive\\Desktop\\example.json";

        // Convert JSON to XML
        XmlDocument xmlDocument = JsonConvert.DeserializeXmlNode(filePath, "Root");

    }
}
