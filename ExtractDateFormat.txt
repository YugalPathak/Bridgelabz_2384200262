using System;
using System.Text.RegularExpressions;

class Program
{
    static void Main()
    {
        // Define a regex pattern to match dates in dd/mm/yyyy format
        string pattern = @"\b(0[1-9]|[12][0-9]|3[01])/(0[1-9]|1[0-2])/\d{4}\b";

        // Sample input text containing dates
        string input = "The events are scheduled for 12/05/2023, 15/08/2024, and 29/02/2020.";

        // Find all matches of dates in the input text
        MatchCollection matches = Regex.Matches(input, pattern);

        // Print extracted dates
        Console.WriteLine("Extracted Dates:");
        foreach (Match match in matches)
        {
            // Print each matched date
            Console.WriteLine(match.Value);
        }
    }
}
