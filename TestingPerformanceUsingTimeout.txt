using System;
using System.Threading;
using NUnit.Framework;

public class PerformanceTester
{
    // Simulates a long-running task that takes 3 seconds
    public void LongRunningTask()
    {
        Thread.Sleep(3000); // Simulating delay
    }
}

[TestFixture]
public class PerformanceTests
{
    private PerformanceTester _tester;

    [SetUp] // Runs before each test
    public void Setup()
    {
        _tester = new PerformanceTester();
    }

    [Test, Timeout(2000)] // Test fails if execution exceeds 2 seconds
    public void LongRunningTask_Should_Complete_Within_Timeout()
    {
        _tester.LongRunningTask();
    }
}
