        // List of test credit card numbers
        string[] cardNumbers = { "4111111111111111", "5111111111111111", "6111111111111111", "4222222222222", "5000111122223333" };

        // Loop through each credit card number and validate it
        foreach (string card in cardNumbers)
        {
            // Print whether the credit card number is valid or invalid
            Console.WriteLine($"{card} â†’ {(Regex.IsMatch(card, pattern) ? "Valid" : "Invalid")}");
        }
    }
}
