
using System;
using System.Text.RegularExpressions;

class Program
{
    static void Main()
    {
        // Define a regex pattern to match capitalized words
        string pattern = @"\b[A-Z][a-z]*\b"; 

        // Sample input text containing capitalized words
        string input = "The Eiffel Tower is in Paris and the Statue of Liberty is in New York.";

        // Find all matches of capitalized words in the input text
        MatchCollection matches = Regex.Matches(input, pattern);

        // Print extracted capitalized words
        Console.WriteLine("Extracted Capitalized Words:");
        foreach (Match match in matches)
        {
            // Print each matched capitalized word followed by a comma
            Console.Write(match.Value + ", ");
        }
    }
}
