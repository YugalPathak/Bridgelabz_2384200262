using NUnit.Framework;

public class NumberChecker
{
    // Method to check if a number is even
    public bool IsEven(int number)
    {
        return number % 2 == 0;
    }
}

[TestFixture]
public class NumberCheckerTests
{
    private NumberChecker _numberChecker;

    [SetUp] // Runs before each test
    public void Setup()
    {
        _numberChecker = new NumberChecker();
    }

    // Parameterized test using [TestCase] with multiple values
    [TestCase(2, true)]
    [TestCase(4, true)]
    [TestCase(6, true)]
    [TestCase(7, false)]
    [TestCase(9, false)]
    public void IsEven_Should_Return_Correct_Result(int number, bool expectedResult)
    {
        bool result = _numberChecker.IsEven(number);
        Assert.AreEqual(expectedResult, result);
    }
}
